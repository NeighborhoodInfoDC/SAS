 Description: Calculate annual percentage change between two numbers, 
 <start> and <end>.

 Modifications:
**************************************************************************/

%macro annchg( start, end, years );
  ( ( (&end) / (&start) ) ** ( 1 / (&years) ) - 1 )
%mend annchg;
