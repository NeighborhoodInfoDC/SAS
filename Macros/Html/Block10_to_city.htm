%macro Block10_to_city( invar=geoblk2010, outvar=city, format=Y );

  length &outvar $ 1;
  
  &outvar = put( &invar, $bk1city. );
  
  label &outvar = "Washington, D.C.";
  
  %if %upcase( &format ) = Y %then %do;
    format &outvar $city.;
  %end;

%mend Block10_to_city;
