%macro Block10_to_tr10( invar=geoblk2010, outvar=Geo2010, format=Y );

  length &outvar $ 11;
  
  &outvar = left( &invar );
  
  label &outvar = "Full census tract ID (2010): ssccctttttt";
  
  %if %upcase( &format ) = Y %then %do;
    format &outvar $geo10a.;
  %end;

%mend Block10_to_tr10;
