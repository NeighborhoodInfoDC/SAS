%macro Octo_Anc2002( invar=Name, outvar=Anc2002 );

  ** ANC code **;
  
  length &outvar $ 2;
  
  if &invar ~= "" then 
    &outvar = ANC_ID;
  else
    &outvar = "";
    
  label
    &outvar = "Advisory Neighborhood Commission (2002)";
    
%mend Octo_Anc2002;
