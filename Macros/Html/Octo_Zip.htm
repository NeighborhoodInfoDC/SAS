%macro Octo_ZIP( invar=Zipcode, outvar=Zip );

  ** ZIP code **;
  
  length &outvar $ 5;
  
  if &invar > 0 then 
    &outvar = put( &invar, z5. );
  else  
    &outvar = "";

  label
    &outvar = "ZIP code (5-digit)";
    
%mend Octo_ZIP;
