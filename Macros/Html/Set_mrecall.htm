 Description: Set MRECALL option based on value of
 _PROGRAM macro var.

%macro Set_mrecall;

  %if &_program = {Interactive} %then %do;
    options mrecall;
  %end;
  %else %do;
    options nomrecall;
  %end;

%mend Set_mrecall;
