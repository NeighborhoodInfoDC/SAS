%macro Tr00_to_cta03( tract=Geo2000, cta=Casey_ta2003 );

  length &cta $ 1;
  
  &cta = put( &tract, $tr0cta. );
  
  label &cta = "Casey target area (2003)";
  
  format &cta $cta03f.;

%mend Tr00_to_cta03;
