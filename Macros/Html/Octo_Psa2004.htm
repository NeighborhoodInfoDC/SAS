%macro Octo_Psa2004( invar=Name, outvar=Psa2004 );

  ** PSA code **;
  
  length &outvar $ 3;
  
  if &invar ~= . then 
    &outvar = put( &invar, z3. );
  else  
    &outvar = "";

  label
    &outvar = "MPD Police Service Area (2004)";
    
%mend Octo_Psa2004;
