%macro Tr10_to_eor( tract=Geo2010, eor=Eor );

  length &eor $ 1;
  
  &eor = put( &tract, $tr1eor. );
  
  label &eor = "East of the Anacostia River";
  
  format &eor $eor.;

%mend Tr10_to_eor;
