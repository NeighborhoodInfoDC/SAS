%macro Tr00_to_eor( tract=Geo2000, eor=Eor );

  length &eor $ 1;
  
  &eor = put( &tract, $tr0eor. );
  
  label &eor = "East of the Anacostia River";
  
  format &eor $eor.;

%mend Tr00_to_eor;
