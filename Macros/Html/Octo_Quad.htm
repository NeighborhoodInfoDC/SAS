%macro Octo_Quad( invar=Quadrant, outvar=Quad );

  length &outvar $ 2;
  
  &outvar = left( upcase( &invar ) );
  
  if &outvar not in ( 'NW', 'NE', 'SW', 'SE' ) then &outvar = '';

  label
    &outvar = "DC quadrant";
    
%mend Octo_Quad;
